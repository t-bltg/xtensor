43 template <class T>
44 std::ptrdiff_t adjust_secondary_stride(std::ptrdiff_t stride, T shape)
45 {
46 return stride != 0 ? stride : static_cast<std::ptrdiff_t>(shape);
47 }
48
49 template <class E>
50 inline std::ptrdiff_t get_secondary_stride(const E& ev)
51 {
52 if (ev.layout() == layout_type::row_major)
53 {
54 return adjust_secondary_stride(ev.strides()[ev.dimension() - 2], *(ev.shape().end() - 1));
55 }
56
57 return adjust_secondary_stride(ev.strides()[1], *(ev.shape().begin()));
58 }
59
60 template <class E>
61 inline std::size_t leading_axis_n_iters(const E& ev)
62 {
63 if (ev.layout() == layout_type::row_major)
64 {
65 return std::accumulate(
66 ev.shape().begin(),
67 ev.shape().end() - 1,
68 std::size_t(1),
69 std::multiplies<>()
70 );
71 }
72 return std::accumulate(ev.shape().begin() + 1, ev.shape().end(), std::size_t(1), std::multiplies<>());
73 }

75 template <class E, class F>
76 inline void call_over_leading_axis(E& ev, F&& fct)
77 {
78 XTENSOR_ASSERT(ev.dimension() >= 2);
79
80 const std::size_t n_iters = leading_axis_n_iters(ev);
81 const std::ptrdiff_t secondary_stride = get_secondary_stride(ev);
82
83 const auto begin = ev.data();
84 const auto end = begin + n_iters * secondary_stride;
85 for (auto iter = begin; iter != end; iter += secondary_stride)
86 {
87 fct(iter, iter + secondary_stride);
88 }
89 }
90
91 template <class E1, class E2, class F>
92 inline void call_over_leading_axis(E1& e1, E2& e2, F&& fct)
93 {
94 XTENSOR_ASSERT(e1.dimension() >= 2);
95 XTENSOR_ASSERT(e1.dimension() == e2.dimension());
96
97 const std::size_t n_iters = leading_axis_n_iters(e1);
98 std::ptrdiff_t const secondary_stride1 = get_secondary_stride(e1);
99 std::ptrdiff_t const secondary_stride2 = get_secondary_stride(e2);
100 XTENSOR_ASSERT(secondary_stride1 == secondary_stride2);
101
102 const auto begin1 = e1.data();
103 const auto end1 = begin1 + n_iters * secondary_stride1;
104 const auto begin2 = e2.data();
105 const auto end2 = begin2 + n_iters * secondary_stride2;
106 auto iter1 = begin1;
107 auto iter2 = begin2;
108 for (; (iter1 != end1) && (iter2 != end2); iter1 += secondary_stride1, iter2 += secondary_stride2)
109 {
110 fct(iter1, iter1 + secondary_stride1, iter2, iter2 + secondary_stride2);
111 }
112 }

114 template <class E>
115 inline std::size_t leading_axis(const E& e)
116 {
117 if (e.layout() == layout_type::row_major)
118 {
119 return e.dimension() - 1;
120 }
121 else if (e.layout() == layout_type::column_major)
122 {
123 return 0;
124 }
125 XTENSOR_THROW(std::runtime_error, "Layout not supported.");
126 }

128 // get permutations to transpose and reverse-transpose array
129 inline std::pair<dynamic_shape<std::size_t>, dynamic_shape<std::size_t>>
130 get_permutations(std::size_t dim, std::size_t ax, layout_type layout)
131 {
132 dynamic_shape<std::size_t> permutation(dim);
133 std::iota(permutation.begin(), permutation.end(), std::size_t(0));
134 permutation.erase(permutation.begin() + std::ptrdiff_t(ax));
135
136 if (layout == layout_type::row_major)
137 {
138 permutation.push_back(ax);
139 }
140 else
141 {
142 permutation.insert(permutation.begin(), ax);
143 }
144
145 // TODO find a more clever way to get reverse permutation?
146 dynamic_shape<std::size_t> reverse_permutation;
147 for (std::size_t i = 0; i < dim; ++i)
148 {
149 auto it = std::find(permutation.begin(), permutation.end(), i);
150 reverse_permutation.push_back(std::size_t(std::distance(permutation.begin(), it)));
151 }
152
153 return std::make_pair(std::move(permutation), std::move(reverse_permutation));
154 }

156 template <class R, class E, class F>
157 inline R map_axis(const E& e, std::ptrdiff_t axis, F&& lambda)
158 {
159 if (e.dimension() == 1)
160 {
161 R res = e;
162 lambda(res.begin(), res.end());
163 return res;
164 }
165
166 std::size_t const ax = normalize_axis(e.dimension(), axis);
167 if (ax == detail::leading_axis(e))
168 {
169 R res = e;
170 detail::call_over_leading_axis(res, std::forward<F>(lambda));
171 return res;
172 }
173
174 dynamic_shape<std::size_t> permutation, reverse_permutation;
175 std::tie(permutation, reverse_permutation) = get_permutations(e.dimension(), ax, e.layout());
176 R res = transpose(e, permutation);
177 detail::call_over_leading_axis(res, std::forward<F>(lambda));
178 res = transpose(res, reverse_permutation);
179 return res;
180 }

271 template <class ConstRandomIt, class RandomIt, class Compare>
272 inline void
273 argsort_iter(ConstRandomIt data_begin, ConstRandomIt data_end, RandomIt idx_begin, RandomIt idx_end, Compare comp)
274 {
275 XTENSOR_ASSERT(std::distance(data_begin, data_end) >= 0);
276 XTENSOR_ASSERT(std::distance(idx_begin, idx_end) == std::distance(data_begin, data_end));
277 (void) idx_end; // TODO(C++17) [[maybe_unused]] only used in assertion.
278
279 std::iota(idx_begin, idx_end, 0);
280 std::sort(
281 idx_begin,
282 idx_end,
283 [&](const auto i, const auto j)
284 {
285 return comp(*(data_begin + i), *(data_begin + j));
286 }
287 );
288 }

362 template <class E, class R = typename detail::linear_argsort_result_type<E>::type>
363 inline auto flatten_argsort_impl(const xexpression<E>& e)
364 {
365 const auto& de = e.derived_cast();
366

368 using const_iterator = decltype(cit);
369 auto ad = xiterator_adaptor<const_iterator, const_iterator>(cit, cit, de.size());
370
371 using result_type = R;
372 result_type result;
373 result.resize({de.size()});
374
375 detail::argsort_iter(de.cbegin(), de.cend(), result.begin(), result.end());
376
377 return result;
378 }

399 template <class E>
400 inline auto argsort(const xexpression<E>& e, std::ptrdiff_t axis = -1)
401 {
402 using eval_type = typename detail::sort_eval_type<E>::type;
403 using result_type = typename detail::argsort_result_type<eval_type>::type;
404
405 const auto& de = e.derived_cast();
406
407 std::size_t ax = normalize_axis(de.dimension(), axis);
408
409 if (de.dimension() == 1)
410 {
411 return detail::flatten_argsort_impl<E, result_type>(e);
412 }
413
414 const auto argsort = [](auto res_begin, auto res_end, auto ev_begin, auto ev_end)
415 {
416 detail::argsort_iter(ev_begin, ev_end, res_begin, res_end);
417 };
418
419 if (ax == detail::leading_axis(de))
420 {
421 result_type res = result_type::from_shape(de.shape());
422 detail::call_over_leading_axis(res, de, argsort);
423 return res;
424 }
425
426 dynamic_shape<std::size_t> permutation, reverse_permutation;
427 std::tie(permutation, reverse_permutation) = detail::get_permutations(de.dimension(), ax, de.layout());
428 eval_type ev = transpose(de, permutation);
429 result_type res = result_type::from_shape(ev.shape());
430 detail::call_over_leading_axis(res, ev, argsort);
431 res = transpose(res, reverse_permutation);
432 return res;
433 }

452 template <class RandomIt, class Iter, class Compare>
453 inline void
454 partition_iter(RandomIt data_begin, RandomIt data_end, Iter kth_begin, Iter kth_end, Compare comp)
455 {
456 XTENSOR_ASSERT(std::distance(data_begin, data_end) >= 0);
457 XTENSOR_ASSERT(std::distance(kth_begin, kth_end) >= 0);
458
459 using idx_type = typename std::iterator_traits<Iter>::value_type;
460
461 idx_type k_last = static_cast<idx_type>(std::distance(data_begin, data_end));
462 for (; kth_begin != kth_end; ++kth_begin)
463 {
464 std::nth_element(data_begin, data_begin + *kth_begin, data_begin + k_last, std::move(comp));
465 k_last = *kth_begin;
466 }
467 }
468
469 template <class RandomIt, class Iter>
470 inline void partition_iter(RandomIt data_begin, RandomIt data_end, Iter kth_begin, Iter kth_end)
471 {
472 return partition_iter(
473 std::move(data_begin),
474 std::move(data_end),
475 std::move(kth_begin),
476 std::move(kth_end),
477 [](const auto& x, const auto& y) -> bool
478 {
479 return x < y;
480 }
481 );
482 }

655 template <class E, class C, class = std::enable_if_t<!xtl::is_integral<C>::value, int>>
656 inline auto argpartition(const xexpression<E>& e, C kth_container, std::ptrdiff_t axis = -1)
657 {
658 using eval_type = typename detail::sort_eval_type<E>::type;
659 using result_type = typename detail::argsort_result_type<eval_type>::type;
660
661 const auto& de = e.derived_cast();
662
663 if (de.dimension() == 1)
664 {
665 return argpartition<E, C, result_type>(e, std::forward<C>(kth_container), xnone());
666 }
667
668 std::sort(kth_container.begin(), kth_container.end());
669 const auto argpartition_w_kth =
670 [&kth_container](auto res_begin, auto res_end, auto ev_begin, auto /*ev_end*/)
671 {
672 std::iota(res_begin, res_end, 0);
673 detail::partition_iter(
674 res_begin,
675 res_end,
676 kth_container.rbegin(),
677 kth_container.rend(),
678 [&ev_begin](auto const& i, auto const& j)
679 {
680 return *(ev_begin + i) < *(ev_begin + j);
681 }
682 );
683 };
684
685 std::size_t const ax = normalize_axis(de.dimension(), axis);
686 if (ax == detail::leading_axis(de))
687 {
688 result_type res = result_type::from_shape(de.shape());
689 detail::call_over_leading_axis(res, de, argpartition_w_kth);
690 return res;
691 }
692
693 dynamic_shape<std::size_t> permutation, reverse_permutation;
694 std::tie(permutation, reverse_permutation) = detail::get_permutations(de.dimension(), ax, de.layout());
695 eval_type ev = transpose(de, permutation);
696 result_type res = result_type::from_shape(ev.shape());
697 detail::call_over_leading_axis(res, ev, argpartition_w_kth);
698 res = transpose(res, reverse_permutation);
699 return res;
700 }

724 template <class S, class I, class K, class O>
725 inline void select_indices_impl(
726 const S& shape,
727 const I& indices,
728 std::size_t axis,
729 std::size_t current_dim,
730 const K& current_index,
731 O& out
732 )
733 {
734 using id_t = typename K::value_type;
735 if ((current_dim < shape.size() - 1) && (current_dim == axis))
736 {
737 for (auto i : indices)
738 {
739 auto idx = current_index;
740 idx[current_dim] = i;
741 select_indices_impl(shape, indices, axis, current_dim + 1, idx, out);
742 }
743 }
744 else if ((current_dim < shape.size() - 1) && (current_dim != axis))
745 {
746 for (id_t i = 0; xtl::cmp_less(i, shape[current_dim]); ++i)
747 {
748 auto idx = current_index;
749 idx[current_dim] = i;
750 select_indices_impl(shape, indices, axis, current_dim + 1, idx, out);
751 }
752 }
753 else if ((current_dim == shape.size() - 1) && (current_dim == axis))
754 {
755 for (auto i : indices)
756 {
757 auto idx = current_index;
758 idx[current_dim] = i;
759 out.push_back(std::move(idx));
760 }
761 }
762 else if ((current_dim == shape.size() - 1) && (current_dim != axis))
763 {
764 for (id_t i = 0; xtl::cmp_less(i, shape[current_dim]); ++i)
765 {
766 auto idx = current_index;
767 idx[current_dim] = i;
768 out.push_back(std::move(idx));
769 }
770 }
771 }

782 // TODO remove when fancy index views are implemented
783 // Poor man's indexing along a single axis as in NumPy a[:, [1, 3, 4]]
784 template <class E, class I>
785 inline auto fancy_indexing(E&& e, const I& indices, std::ptrdiff_t axis)
786 {
787 std::size_t const ax = normalize_axis(e.dimension(), axis);
788 using shape_t = get_strides_t<typename std::decay_t<E>::shape_type>;
789 auto shape = xtl::forward_sequence<shape_t, decltype(e.shape())>(e.shape());
790 shape[ax] = indices.size();

797 template <class T, class I, class P>
798 inline auto quantile_kth_gamma(std::size_t n, const P& probas, T alpha, T beta)
799 {
800 const auto m = alpha + probas * (T(1) - alpha - beta);
801 // Evaluting since reused a lot
802 const auto p_n_m = eval(probas * static_cast<T>(n) + m - 1);
803 // Previous (virtual) index, may be out of bounds
804 const auto j = floor(p_n_m);
805 const auto j_jp1 = concatenate(xtuple(j, j + 1));
806 // Both interpolation indices, k and k+1
807 const auto k_kp1 = xt::cast<std::size_t>(clip(j_jp1, T(0), T(n - 1)));
808 // Both interpolation coefficients, 1-gamma and gamma
809 const auto omg_g = concatenate(xtuple(T(1) - (p_n_m - j), p_n_m - j));
810 return std::make_pair(eval(k_kp1), eval(omg_g));
811 }

853 template <class T = double, class E, class P>
854 inline auto quantile(E&& e, const P& probas, std::ptrdiff_t axis, T alpha, T beta)
855 {
856 XTENSOR_ASSERT(all(0. <= probas));
857 XTENSOR_ASSERT(all(probas <= 1.));
858 XTENSOR_ASSERT(0. <= alpha);
859 XTENSOR_ASSERT(alpha <= 1.);
860 XTENSOR_ASSERT(0. <= beta);
861 XTENSOR_ASSERT(beta <= 1.);
862
863 using tmp_shape_t = get_strides_t<typename std::decay_t<E>::shape_type>;
864 using id_t = typename tmp_shape_t::value_type;
865
866 std::size_t const ax = normalize_axis(e.dimension(), axis);
867 std::size_t const n = e.shape()[ax];
868 auto kth_gamma = detail::quantile_kth_gamma<T, id_t, P>(n, probas, alpha, beta);
869
870 // Select relevant values for computing interpolating quantiles
871 auto e_partition = xt::partition(std::forward<E>(e), kth_gamma.first, ax);
872 auto e_kth = detail::fancy_indexing(std::move(e_partition), std::move(kth_gamma.first), ax);
873
874 // Reshape interpolation coefficients
875 auto gm1_g_shape = xtl::make_sequence<tmp_shape_t>(e.dimension(), 1);
876 gm1_g_shape[ax] = kth_gamma.second.size();
877 auto gm1_g_reshaped = reshape_view(std::move(kth_gamma.second), std::move(gm1_g_shape));
878
879 // Compute interpolation
880 // TODO(C++20) use (and create) xt::lerp in C++
881 auto e_kth_g = std::move(e_kth) * std::move(gm1_g_reshaped);
882 // Reshape pairwise interpolate for suming along new axis
883 auto e_kth_g_shape = detail::unsqueeze_shape(e_kth_g.shape(), ax);
884 e_kth_g_shape[ax] = 2;
885 e_kth_g_shape[ax + 1] /= 2;
886 auto quantiles = xt::sum(reshape_view(std::move(e_kth_g), std::move(e_kth_g_shape)), ax);
887 // Cannot do a transpose on a non-strided expression so we have to eval
888 return moveaxis(eval(std::move(quantiles)), ax, 0);
889 }

1113 inline typename argfunc_result_type<E>::type arg_func_impl(const E& e, std::size_t axis, F&& cmp)
1114 {
1115 using eval_type = typename detail::sort_eval_type<E>::type;
1116 using value_type = typename E::value_type;
1117 using result_type = typename argfunc_result_type<E>::type;
1118 using result_shape_type = typename result_type::shape_type;
1119
1120 if (e.dimension() == 1)
1121 {
1122 auto begin = e.template begin<L>();
1123 auto end = e.template end<L>();
1124 std::size_t i = static_cast<std::size_t>(std::distance(begin, std::min_element(begin, end)));
1125 return xtensor<size_t, 0>{i};
1126 }
1127
1128 result_shape_type alt_shape;
1129 xt::resize_container(alt_shape, e.dimension() - 1);
1130
1131 // Excluding copy, copy all of shape except for axis
1132 std::copy(e.shape().cbegin(), e.shape().cbegin() + std::ptrdiff_t(axis), alt_shape.begin());
1133 std::copy(
1134 e.shape().cbegin() + std::ptrdiff_t(axis) + 1,
1135 e.shape().cend(),
1136 alt_shape.begin() + std::ptrdiff_t(axis)
1137 );
1138
1139 result_type result = result_type::from_shape(std::move(alt_shape));
1140 auto result_iter = result.template begin<L>();
1141
1142 auto arg_func_lambda = [&result_iter, &cmp](auto begin, auto end)
1143 {
1144 std::size_t idx = 0;
1145 value_type val = *begin;
1146 ++begin;
1147 for (std::size_t i = 1; begin != end; ++begin, ++i)
1148 {
1149 if (cmp(*begin, val))
1150 {
1151 val = *begin;
1152 idx = i;
1153 }
1154 }
1155 *result_iter = idx;
1156 ++result_iter;
1157 };
1158
1159 if (axis != detail::leading_axis(e))
1160 {
1161 dynamic_shape<std::size_t> permutation, reverse_permutation;
1162 std::tie(
1163 permutation,
1164 reverse_permutation
1165 ) = detail::get_permutations(e.dimension(), axis, e.layout());
1166
1167 // note: creating copy
1168 eval_type input = transpose(e, permutation);
1169 detail::call_over_leading_axis(input, arg_func_lambda);
1170 return result;
1171 }
1172 else
1173 {
1174 auto&& input = eval(e);
1175 detail::call_over_leading_axis(input, arg_func_lambda);
1176 return result;
1177 }
1178 }
